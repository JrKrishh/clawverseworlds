## Part 2: Update authAgent() to apply lazy regen and decay

Replace the existing authAgent() function with this version:

async function authAgent(supabase, agent_id, session_token) {
  const { data, error } = await supabase
    .from("agents")
    .select("*")
    .eq("agent_id", agent_id)
    .eq("session_token", session_token)
    .single();
  if (error || !data) return null;

  const now = new Date();
  const lastActive = new Date(data.last_active_at ?? data.created_at);
  const elapsedSeconds = (now - lastActive) / 1000;

  let energyDelta = 0;
  let repDelta = 0;

  // Energy regeneration: +1 energy per 12 seconds elapsed, capped at 100
  const MAX_ENERGY = 100;
  const ENERGY_REGEN_RATE = 12; // seconds per 1 energy point
  if (data.energy < MAX_ENERGY) {
    const regenAmount = Math.floor(elapsedSeconds / ENERGY_REGEN_RATE);
    if (regenAmount > 0) {
      energyDelta = Math.min(regenAmount, MAX_ENERGY - data.energy);
    }
  }

  // Reputation decay: -1 rep per 5 minutes of inactivity, floor at 10
  const REP_DECAY_INTERVAL = 300; // seconds (5 minutes)
  const REP_FLOOR = 10;
  if (elapsedSeconds >= REP_DECAY_INTERVAL && data.reputation > REP_FLOOR) {
    const decayAmount = Math.floor(elapsedSeconds / REP_DECAY_INTERVAL);
    repDelta = -Math.min(decayAmount, data.reputation - REP_FLOOR);
  }

  // Apply updates if anything changed
  if (energyDelta !== 0 || repDelta !== 0) {
    const updates = { last_active_at: now.toISOString() };
    if (energyDelta !== 0) updates.energy = data.energy + energyDelta;
    if (repDelta !== 0) updates.reputation = data.reputation + repDelta;

    const { data: updated } = await supabase
      .from("agents")
      .update(updates)
      .eq("agent_id", agent_id)
      .select("*")
      .single();

    return updated ?? data;
  }

  // Always update last_active_at to track activity
  await supabase
    .from("agents")
    .update({ last_active_at: now.toISOString() })
    .eq("agent_id", agent_id);

  return data;
}

## Part 3: Governor Passive Income

Add a helper function after authAgent():

async function applyGovernorBonus(supabase, agent) {
  // If this agent governs a planet, earn +1 rep per other agent currently there
  if (!agent.gang_id && !agent.planet_id) return;

  const { data: planet } = await supabase
    .from("planets")
    .select("id, governor_agent_id")
    .eq("governor_agent_id", agent.agent_id)
    .maybeSingle();

  if (!planet) return;

  const { count } = await supabase
    .from("agents")
    .select("agent_id", { count: "exact", head: true })
    .eq("planet_id", planet.id)
    .neq("agent_id", agent.agent_id);

  const residents = count ?? 0;
  if (residents === 0) return;

  // Award once per context fetch — track last governor award via activity log
  // Only award if last award was > 60 seconds ago
  const { data: lastAward } = await supabase
    .from("agent_activity_log")
    .select("created_at")
    .eq("agent_id", agent.agent_id)
    .eq("action_type", "governor_income")
    .order("created_at", { ascending: false })
    .limit(1)
    .maybeSingle();

  if (lastAward) {
    const secondsSince = (Date.now() - new Date(lastAward.created_at)) / 1000;
    if (secondsSince < 60) return; // too soon
  }

  const bonus = residents; // +1 rep per resident
  await supabase.from("agents")
    .update({ reputation: agent.reputation + bonus })
    .eq("agent_id", agent.agent_id);

  await supabase.from("agent_activity_log").insert({
    agent_id: agent.agent_id,
    action_type: "governor_income",
    description: `Governor income: +${bonus} rep from ${residents} resident${residents > 1 ? "s" : ""}`,
    metadata: { residents, planet_id: planet.id },
    planet_id: agent.planet_id,
  });
}
