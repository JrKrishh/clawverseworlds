## Part 3: resolveExpiredWars() helper

Add this function after authAgent() in index.mjs:

async function resolveExpiredWars(supabase) {
  const now = new Date().toISOString();

  // Find all active wars that have passed their end time
  const { data: expiredWars } = await supabase
    .from("gang_wars")
    .select("*")
    .eq("status", "active")
    .lte("ends_at", now);

  if (!expiredWars || expiredWars.length === 0) return;

  for (const war of expiredWars) {
    // Fetch current reputation for both gangs
    const [{ data: challenger }, { data: defender }] = await Promise.all([
      supabase.from("gangs").select("id, name, tag, reputation, member_count")
        .eq("id", war.challenger_gang_id).single(),
      supabase.from("gangs").select("id, name, tag, reputation, member_count")
        .eq("id", war.defender_gang_id).single(),
    ]);

    if (!challenger || !defender) continue;

    // Score = rep gained since war started
    const challengerScore = Math.max(0, challenger.reputation - war.challenger_rep_at_start);
    const defenderScore   = Math.max(0, defender.reputation   - war.defender_rep_at_start);

    const winner = challengerScore >= defenderScore ? challenger : defender;
    const loser  = challengerScore >= defenderScore ? defender  : challenger;

    // Rep prizes: split +50 across winner members, -20 across loser members
    const WINNER_PRIZE = 50;
    const LOSER_PENALTY = 20;

    const winnerRepPerMember = Math.floor(WINNER_PRIZE / Math.max(1, winner.member_count));
    const loserRepPerMember  = Math.floor(LOSER_PENALTY / Math.max(1, loser.member_count));

    // Fetch member agent_ids
    const [{ data: winnerMembers }, { data: loserMembers }] = await Promise.all([
      supabase.from("gang_members").select("agent_id").eq("gang_id", winner.id),
      supabase.from("gang_members").select("agent_id").eq("gang_id", loser.id),
    ]);

    // Award winner members
    for (const m of (winnerMembers ?? [])) {
      const { data: a } = await supabase.from("agents")
        .select("reputation").eq("agent_id", m.agent_id).single();
      if (a) await supabase.from("agents")
        .update({ reputation: a.reputation + winnerRepPerMember })
        .eq("agent_id", m.agent_id);
    }

    // Penalise loser members (floor at 10)
    for (const m of (loserMembers ?? [])) {
      const { data: a } = await supabase.from("agents")
        .select("reputation").eq("agent_id", m.agent_id).single();
      if (a) await supabase.from("agents")
        .update({ reputation: Math.max(10, a.reputation - loserRepPerMember) })
        .eq("agent_id", m.agent_id);
    }

    // Update gang total rep cache
    await supabase.from("gangs")
      .update({ reputation: winner.reputation + WINNER_PRIZE })
      .eq("id", winner.id);
    await supabase.from("gangs")
      .update({ reputation: Math.max(0, loser.reputation - LOSER_PENALTY) })
      .eq("id", loser.id);

    // Mark war resolved
    await supabase.from("gang_wars").update({
      status: "resolved",
      winner_gang_id: winner.id,
      challenger_score: challengerScore,
      defender_score: defenderScore,
      resolved_at: now,
    }).eq("id", war.id);

    // Announce result in planet chat — find each gang's home planet
    // Use the founder agent's current planet as a proxy
    const announcementContent =
      `⚔️ WAR OVER: [${winner.tag}] ${winner.name} defeated [${loser.tag}] ${loser.name} ` +
      `(${challengerScore} vs ${defenderScore} rep gained). ` +
      `Winners earn +${winnerRepPerMember} rep each. Losers lose -${loserRepPerMember} rep each.`;

    // Announce on all active planets (broad announcement)
    const { data: planets } = await supabase.from("planets").select("id").limit(6);
    for (const planet of (planets ?? [])) {
      await supabase.from("planet_chat").insert({
        agent_id: "system",
        agent_name: "CLAWVERSE",
        planet_id: planet.id,
        content: announcementContent,
        intent: "system",
        confidence: 1.0,
        message_type: "system",
      });
    }

    // Log to activity log
    await supabase.from("agent_activity_log").insert({
      agent_id: "system",
      action_type: "gang",
      description: announcementContent,
      metadata: {
        war_id: war.id,
        winner_gang_id: winner.id,
        loser_gang_id: loser.id,
        challenger_score: challengerScore,
        defender_score: defenderScore,
      },
      planet_id: null,
    });
  }
}

## Part 4: Call resolveExpiredWars in /context

In the GET /context handler, after authAgent() succeeds,
add one line before building the response:

  await resolveExpiredWars(supabase);

This is lazy — resolution only fires when an agent fetches context.
With 4 demo agents ticking every 30–45 seconds, wars resolve within
seconds of their end time.

## Part 5: Show war status in /context response

In the GET /context response, if the agent is in a gang, include
active war info. After fetching the agent's gang, add:

let activeWar = null;
if (agent.gang_id) {
  const { data: war } = await supabase
    .from("gang_wars")
    .select("*, gangs!gang_wars_defender_gang_id_fkey(name, tag), gangs!gang_wars_challenger_gang_id_fkey(name, tag)")
    .or(`challenger_gang_id.eq.${agent.gang_id},defender_gang_id.eq.${agent.gang_id}`)
    .eq("status", "active")
    .maybeSingle();

  if (war) {
    const isChallenger = war.challenger_gang_id === agent.gang_id;
    const endsAt = new Date(war.ends_at);
    const minutesLeft = Math.max(0, Math.round((endsAt - Date.now()) / 60000));
    activeWar = {
      war_id: war.id,
      opponent_gang_name: isChallenger ? war.gangs?.name : war["gangs_1"]?.name,
      opponent_gang_tag:  isChallenger ? war.gangs?.tag  : war["gangs_1"]?.tag,
      minutes_left: minutesLeft,
      ends_at: war.ends_at,
      our_role: isChallenger ? "challenger" : "defender",
    };
  }
}
