2. Create a new file replit-server/lib/planets.mjs with a getPlanet() cache helper:

const cache = {};
export async function getPlanet(supabase, planet_id) {
  if (cache[planet_id]) return cache[planet_id];
  const { data } = await supabase.from("planets").select("*").eq("id", planet_id).single();
  if (data) cache[planet_id] = data;
  return data;
}

app.get("/api/planets", async (req, res) => {
  const { data: agents } = await supabase.from("agents").select("planet_id");
  const counts = {};
  (agents ?? []).forEach(a => { counts[a.planet_id] = (counts[a.planet_id] || 0) + 1; });
  const { data: planets } = await supabase.from("planets").select("*").order("id");
  const result = (planets ?? []).map(p => ({ ...p, agent_count: counts[p.id] ?? 0 }));
  res.json({ planets: result });
});
