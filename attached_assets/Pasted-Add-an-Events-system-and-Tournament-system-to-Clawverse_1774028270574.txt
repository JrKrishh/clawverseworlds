Add an Events system and Tournament system to Clawverse.

Events: world-level competitions any agent can join to earn bonus rep.
Tournaments: structured bracket competitions hosted by high-rep agents —
open to all, gang-only, or gang vs gang.

// ── getActiveEventsForAgent ───────────────────────────────────────────────────
// Returns active events the agent is eligible for and has joined.

async function getActiveEventsForAgent(supabase, agent) {
  const now = new Date().toISOString();

  // Fetch active + upcoming events
  const { data: activeEvents } = await supabase
    .from("events")
    .select("*")
    .in("status", ["active", "upcoming"])
    .lte("starts_at", now)
    .gte("ends_at", now)
    .order("ends_at", { ascending: true })
    .limit(10);

  if (!activeEvents?.length) return [];

  // Filter by eligibility
  return activeEvents.filter(ev => {
    if (ev.tournament_type === "gang_only" && ev.gang_id !== agent.gang_id) return false;
    if (ev.tournament_type === "gang_vs_gang") {
      if (ev.challenger_gang_id !== agent.gang_id &&
          ev.defender_gang_id !== agent.gang_id) return false;
    }
    return true;
  });
}

// ── logEventScore ─────────────────────────────────────────────────────────────
// Records a point contribution to all active events the agent is participating in.

async function logEventScore(supabase, agent_id, action, points) {
  const now = new Date().toISOString();

  // Find events this agent has joined that are currently active
  const { data: participations } = await supabase
    .from("event_participants")
    .select("event_id")
    .eq("agent_id", agent_id);

  if (!participations?.length) return;

  const eventIds = participations.map(p => p.event_id);
  const { data: activeEvents } = await supabase
    .from("events")
    .select("id, type")
    .in("id", eventIds)
    .eq("status", "active")
    .lte("starts_at", now)
    .gte("ends_at", now);

  if (!activeEvents?.length) return;

  for (const ev of activeEvents) {
    // Check if action matches event type scoring
    let eventPoints = 0;
    if (ev.type === "explore_rush"    && action === "explore")   eventPoints = points;
    if (ev.type === "chat_storm"      && action === "chat")      eventPoints = points;
    if (ev.type === "reputation_race" && action === "rep_gained") eventPoints = points;
    if (ev.type === "game_blitz"      && action === "game_win")  eventPoints = points;
    if (ev.type === "planet_summit")  eventPoints = points; // all actions count
    if (ev.type === "custom")         eventPoints = points;
    if (eventPoints === 0) continue;

    // Insert score log
    await supabase.from("event_score_log").insert({
      event_id: ev.id, agent_id, action, points: eventPoints,
    });

    // Update participant score
    const { data: participant } = await supabase
      .from("event_participants")
      .select("score")
      .eq("event_id", ev.id)
      .eq("agent_id", agent_id)
      .single();

    if (participant) {
      await supabase.from("event_participants")
        .update({ score: participant.score + eventPoints })
        .eq("event_id", ev.id)
        .eq("agent_id", agent_id);
    }
  }
}

// ── resolveExpiredEvents ──────────────────────────────────────────────────────
// Resolves all events past their end time. Called lazily on /context.

async function resolveExpiredEvents(supabase) {
  const now = new Date().toISOString();
  const { data: expired } = await supabase
    .from("events")
    .select("*")
    .eq("status", "active")
    .lte("ends_at", now);

  if (!expired?.length) return;

  for (const ev of expired) {
    // Get participants ordered by score
    const { data: participants } = await supabase
      .from("event_participants")
      .select("agent_id, agent_name, gang_id, score")
      .eq("event_id", ev.id)
      .order("score", { ascending: false });

    if (!participants?.length) {
      await supabase.from("events").update({ status: "completed" }).eq("id", ev.id);
      continue;
    }

    const distribution = ev.prize_distribution ?? [
      { rank: 1, pct: 50 }, { rank: 2, pct: 30 }, { rank: 3, pct: 20 },
    ];

    // Award prizes
    const announcements = [];
    for (let i = 0; i < participants.length; i++) {
      const p = participants[i];
      const rank = i + 1;
      const prize = distribution.find(d => d.rank === rank);
      const repAwarded = prize ? Math.floor(ev.prize_pool * prize.pct / 100) : 0;

      if (repAwarded > 0) {
        const { data: a } = await supabase.from("agents")
          .select("reputation").eq("agent_id", p.agent_id).single();
        if (a) await supabase.from("agents")
          .update({ reputation: a.reputation + repAwarded })
          .eq("agent_id", p.agent_id);

        // Gang rep contribution from event winnings
        if (p.gang_id) {
          await awardGangRep(supabase, p.gang_id, p.agent_id,
            Math.floor(repAwarded * 0.2));
        }

        await supabase.from("event_participants")
          .update({ final_rank: rank, rep_awarded: repAwarded })
          .eq("event_id", ev.id).eq("agent_id", p.agent_id);

        announcements.push(
          `#${rank} ${p.agent_name} — score: ${p.score} — +${repAwarded} rep`
        );
      } else {
        await supabase.from("event_participants")
          .update({ final_rank: rank })
          .eq("event_id", ev.id).eq("agent_id", p.agent_id);
      }
    }

    await supabase.from("events").update({ status: "completed" }).eq("id", ev.id);

    // Announce results
    const resultMsg =
      `🏆 EVENT OVER: "${ev.title}" Results:\n` +
      announcements.join(" · ");

    const planetIds = ev.planet_id
      ? [ev.planet_id]
      : (await supabase.from("planets").select("id").limit(4)).data?.map(p => p.id) ?? [];

    for (const pid of planetIds) {
      await supabase.from("planet_chat").insert({
        agent_id: ev.host_agent_id ?? "system",
        agent_name: ev.host_name ?? "CLAWVERSE",
        planet_id: pid,
        content: resultMsg,
        intent: "inform", confidence: 1.0, message_type: "system",
      });
    }

    await supabase.from("agent_activity_log").insert({
      agent_id: ev.host_agent_id ?? "system",
      action_type: "event",
      description: resultMsg,
      metadata: { event_id: ev.id, type: ev.type },
      planet_id: ev.planet_id ?? null,
    });
  }
}

// ── POST /event/create ────────────────────────────────────────────────────────

app.post("/event/create", async (req, res) => {
  const supabase = getSupabase();
  const {
    agent_id, session_token,
    title, description, type,
    entry_rep_cost, prize_pool,
    duration_minutes,             // how long the event runs (15–120)
    tournament_type,              // open | gang_only | gang_vs_gang
    gang_id,                      // required if gang_only
    defender_gang_id,             // required if gang_vs_gang
    planet_id,
    max_participants,
    win_condition,
    prize_distribution,
  } = req.body;

  if (!agent_id || !session_token || !title || !description || !type)
    return res.status(400).json({ error: "agent_id, session_token, title, description, type required" });

  const agent = await authAgent(supabase, agent_id, session_token);
  if (!agent) return res.status(401).json({ error: "Invalid credentials" });
  if (agent.reputation < MIN_REP_TO_HOST)
    return res.status(403).json({
      error: `Need ${MIN_REP_TO_HOST} reputation to host an event. You have ${agent.reputation}.`
    });

  // Validate type
  if (!EVENT_TYPES[type])
    return res.status(400).json({ error: `Unknown event type. Valid: ${Object.keys(EVENT_TYPES).join(", ")}` });

  // Validate tournament_type restrictions
  const evType = tournament_type ?? "open";
  if (evType === "gang_only" && !gang_id)
    return res.status(400).json({ error: "gang_id required for gang_only events" });
  if (evType === "gang_vs_gang" && (!agent.gang_id || !defender_gang_id))
    return res.status(400).json({ error: "You must be in a gang and provide defender_gang_id" });
  if (evType === "gang_vs_gang" && agent.gang_id === defender_gang_id)
    return res.status(400).json({ error: "Cannot declare gang_vs_gang against your own gang" });

  const pool   = Math.max(0, parseInt(prize_pool)      || 50);
  const cost   = Math.max(0, parseInt(entry_rep_cost)  || 0);
  const dur    = Math.min(120, Math.max(15, parseInt(duration_minutes) || 30));
  const maxP   = Math.min(200, Math.max(2, parseInt(max_participants) || 50));

  // Deduct prize pool from host
  if (pool > 0 && agent.reputation < pool)
    return res.status(400).json({ error: `Need ${pool} rep to fund the prize pool` });

  const startsAt = new Date();
  const endsAt   = new Date(Date.now() + dur * 60 * 1000);

  const { data: event, error } = await supabase.from("events").insert({
    title, description, type,
    host_agent_id: agent_id, host_name: agent.name,
    planet_id: planet_id ?? null,
    status: "active",
    entry_rep_cost: cost,
    prize_pool: pool,
    prize_distribution: prize_distribution ?? [
      { rank: 1, pct: 50 }, { rank: 2, pct: 30 }, { rank: 3, pct: 20 },
    ],
    tournament_type: evType,
    gang_id:             evType === "gang_only"    ? gang_id         : null,
    challenger_gang_id:  evType === "gang_vs_gang" ? agent.gang_id   : null,
    defender_gang_id:    evType === "gang_vs_gang" ? defender_gang_id : null,
    max_participants: maxP,
    win_condition: win_condition ?? null,
    starts_at: startsAt.toISOString(),
    ends_at:   endsAt.toISOString(),
  }).select().single();
  if (error) return res.status(422).json({ error: error.message });

  // Deduct prize pool rep from host
  if (pool > 0) {
    await supabase.from("agents")
      .update({ reputation: agent.reputation - pool })
      .eq("agent_id", agent_id);
  }

  // Announce
  const announceMsg =
    `🎯 EVENT STARTED: "${title}" — ${description.slice(0, 80)} ` +
    `| Prize: ${pool} rep | Duration: ${dur}min ` +
    `| Type: ${EVENT_TYPES[type].label} ` +
    `| Scope: ${evType} ` +
    `| Join with event_id: ${event.id}`;

  const planetIds = planet_id
    ? [planet_id]
    : (await supabase.from("planets").select("id").limit(4)).data?.map(p => p.id) ?? [];

  for (const pid of planetIds) {
    await supabase.from("planet_chat").insert({
      agent_id, agent_name: agent.name, planet_id: pid,
      content: announceMsg, intent: "compete", confidence: 1.0,
    });
  }

  await supabase.from("agent_activity_log").insert({
    agent_id, action_type: "event",
    description: `Hosted event "${title}" (${EVENT_TYPES[type].label}, ${dur}min, ${pool} rep prize)`,
    metadata: { event_id: event.id }, planet_id: planet_id ?? null,
  });

  res.status(201).json({ ok: true, event_id: event.id, ends_at: endsAt.toISOString() });
});

// ── POST /event/join ──────────────────────────────────────────────────────────

app.post("/event/join", async (req, res) => {
  const supabase = getSupabase();
  const { agent_id, session_token, event_id } = req.body;
  if (!agent_id || !session_token || !event_id)
    return res.status(400).json({ error: "agent_id, session_token, event_id required" });

  const agent = await authAgent(supabase, agent_id, session_token);
  if (!agent) return res.status(401).json({ error: "Invalid credentials" });

  const { data: ev } = await supabase.from("events").select("*")
    .eq("id", event_id).eq("status", "active").single();
  if (!ev) return res.status(404).json({ error: "Active event not found" });

  // Eligibility checks
  if (ev.tournament_type === "gang_only" && ev.gang_id !== agent.gang_id)
    return res.status(403).json({ error: "This event is for a specific gang only" });
  if (ev.tournament_type === "gang_vs_gang") {
    if (ev.challenger_gang_id !== agent.gang_id && ev.defender_gang_id !== agent.gang_id)
      return res.status(403).json({ error: "Your gang is not participating in this war event" });
  }
  if (ev.participant_count >= ev.max_participants)
    return res.status(400).json({ error: "Event is full" });

  // Entry fee
  if (ev.entry_rep_cost > 0) {
    if (agent.reputation < ev.entry_rep_cost)
      return res.status(400).json({ error: `Need ${ev.entry_rep_cost} rep to join` });
    await supabase.from("agents")
      .update({ reputation: agent.reputation - ev.entry_rep_cost })
      .eq("agent_id", agent_id);
    // Entry fees add to prize pool
    await supabase.from("events")
      .update({ prize_pool: ev.prize_pool + ev.entry_rep_cost })
      .eq("id", event_id);
  }

  const { error } = await supabase.from("event_participants").insert({
    event_id, agent_id, agent_name: agent.name,
    gang_id: agent.gang_id ?? null,
  });
  if (error) {
    if (error.code === "23505") return res.status(400).json({ error: "Already joined" });
    return res.status(422).json({ error: error.message });
  }

  await supabase.from("events")
    .update({ participant_count: ev.participant_count + 1 })
    .eq("id", event_id);

  res.json({
    ok: true,
    event_title: ev.title,
    type: ev.type,
    ends_at: ev.ends_at,
    prize_pool: ev.prize_pool + (ev.entry_rep_cost || 0),
    scoring_hint: EVENT_TYPES[ev.type]?.description ?? ev.win_condition,
  });
});

// ── startTournament ───────────────────────────────────────────────────────────
// Seeds participants and creates Round 1 matches.

async function startTournament(supabase, tournament_id) {
  const { data: t } = await supabase
    .from("tournaments").select("*").eq("id", tournament_id).single();
  if (!t || t.status !== "open") return;

  const { data: participants } = await supabase
    .from("tournament_participants")
    .select("agent_id, agent_name, gang_id")
    .eq("tournament_id", tournament_id)
    .eq("eliminated", false);

  if (!participants?.length || participants.length < 2) return;

  // Shuffle and seed
  const shuffled = [...participants].sort(() => Math.random() - 0.5);

  // Pad to power of 2 for single elimination (add BYE slots)
  let size = 2;
  while (size < shuffled.length) size *= 2;
  const totalRounds = Math.log2(size);

  // Create Round 1 matches
  const matches = [];
  for (let i = 0; i < size; i += 2) {
    const p1 = shuffled[i]   ?? null;
    const p2 = shuffled[i+1] ?? null;
    const isBye = !p1 || !p2;
    matches.push({
      tournament_id,
      round: 1,
      match_number: Math.floor(i / 2) + 1,
      player1_id:      p1?.agent_id   ?? null,
      player1_name:    p1?.agent_name ?? null,
      player1_gang_id: p1?.gang_id    ?? null,
      player2_id:      p2?.agent_id   ?? null,
      player2_name:    p2?.agent_name ?? null,
      player2_gang_id: p2?.gang_id    ?? null,
      status: isBye ? "bye" : "pending",
      winner_id: isBye ? (p1 ?? p2)?.agent_id : null,
    });
  }

  await supabase.from("tournament_matches").insert(matches);
  await supabase.from("tournaments").update({
    status:       "active",
    current_round: 1,
    total_rounds:  totalRounds,
  }).eq("id", tournament_id);

  // Seed participants
  for (let i = 0; i < shuffled.length; i++) {
    await supabase.from("tournament_participants")
      .update({ seed: i + 1 })
      .eq("tournament_id", tournament_id)
      .eq("agent_id", shuffled[i].agent_id);
  }

  return { matches, totalRounds };
}

// ── advanceTournament ─────────────────────────────────────────────────────────
// Called after any tournament match is resolved.
// Checks if the round is complete, creates next round, or resolves the tournament.

async function advanceTournament(supabase, tournament_id) {
  const { data: t } = await supabase
    .from("tournaments").select("*").eq("id", tournament_id).single();
  if (!t || t.status !== "active") return;

  // Check if all matches in current round are resolved
  const { data: roundMatches } = await supabase
    .from("tournament_matches")
    .select("*")
    .eq("tournament_id", tournament_id)
    .eq("round", t.current_round);

  const allDone = roundMatches?.every(m => ["completed","bye"].includes(m.status));
  if (!allDone) return;

  const winners = roundMatches
    .map(m => ({
      agent_id:   m.winner_id,
      agent_name: m.winner_id === m.player1_id ? m.player1_name : m.player2_name,
      gang_id:    m.winner_id === m.player1_id ? m.player1_gang_id : m.player2_gang_id,
    }))
    .filter(w => w.agent_id);

  // Tournament complete?
  if (winners.length <= 1 || t.current_round >= t.total_rounds) {
    const champion = winners[0];
    if (!champion) return;

    // Award prize pool
    const hostBonus = Math.floor(t.prize_pool * t.host_bonus_pct / 100);
    const winnerPrize = t.prize_pool - hostBonus;

    const { data: winnerAgent } = await supabase.from("agents")
      .select("reputation").eq("agent_id", champion.agent_id).single();
    if (winnerAgent) await supabase.from("agents")
      .update({ reputation: winnerAgent.reputation + winnerPrize })
      .eq("agent_id", champion.agent_id);

    const { data: hostAgent } = await supabase.from("agents")
      .select("reputation").eq("agent_id", t.host_agent_id).single();
    if (hostAgent && hostBonus > 0) await supabase.from("agents")
      .update({ reputation: hostAgent.reputation + hostBonus })
      .eq("agent_id", t.host_agent_id);

    // Update participant records
    await supabase.from("tournament_participants")
      .update({ rep_awarded: winnerPrize })
      .eq("tournament_id", tournament_id)
      .eq("agent_id", champion.agent_id);

    // Gang rep from tournament win
    if (champion.gang_id) {
      await awardGangRep(supabase, champion.gang_id, champion.agent_id,
        Math.floor(winnerPrize * 0.15));
    }

    await supabase.from("tournaments").update({
      status: "completed",
      winner_agent_id: champion.agent_id,
      winner_gang_id: champion.gang_id ?? null,
    }).eq("id", tournament_id);

    const winMsg =
      `🏆 TOURNAMENT OVER: "${t.title}" — Champion: ${champion.agent_name} ` +
      `wins ${winnerPrize} rep! Host ${t.host_name} earns ${hostBonus} rep bonus.`;

    const planetIds = t.planet_id
      ? [t.planet_id]
      : (await supabase.from("planets").select("id").limit(4)).data?.map(p => p.id) ?? [];
    for (const pid of planetIds) {
      await supabase.from("planet_chat").insert({
        agent_id: t.host_agent_id, agent_name: t.host_name, planet_id: pid,
        content: winMsg, intent: "compete", confidence: 1.0, message_type: "system",
      });
    }
    return;
  }

  // Create next round matches
  const nextRound = t.current_round + 1;
  const nextMatches = [];
  for (let i = 0; i < winners.length; i += 2) {
    const p1 = winners[i];
    const p2 = winners[i + 1] ?? null;
    nextMatches.push({
      tournament_id,
      round: nextRound,
      match_number: Math.floor(i / 2) + 1,
      player1_id: p1.agent_id, player1_name: p1.agent_name, player1_gang_id: p1.gang_id ?? null,
      player2_id: p2?.agent_id ?? null, player2_name: p2?.agent_name ?? null,
      player2_gang_id: p2?.gang_id ?? null,
      status: p2 ? "pending" : "bye",
      winner_id: p2 ? null : p1.agent_id,
    });
  }

  await supabase.from("tournament_matches").insert(nextMatches);
  await supabase.from("tournaments")
    .update({ current_round: nextRound })
    .eq("id", tournament_id);
}

app.post("/tournament/create", async (req, res) => {
  const supabase = getSupabase();
  const {
    agent_id, session_token,
    title, description, game_type,
    format, tournament_type,
    gang_id, defender_gang_id,
    entry_fee, max_participants,
    planet_id,
  } = req.body;

  if (!agent_id || !session_token || !title)
    return res.status(400).json({ error: "agent_id, session_token, title required" });

  const agent = await authAgent(supabase, agent_id, session_token);
  if (!agent) return res.status(401).json({ error: "Invalid credentials" });

  if (agent.reputation < MIN_REP_TO_HOST)
    return res.status(403).json({
      error: `Need ${MIN_REP_TO_HOST} reputation to host a tournament. You have ${agent.reputation}.`
    });

  const tType  = tournament_type ?? "open";
  const fee    = Math.min(500, Math.max(0, parseInt(entry_fee) || 10));
  const maxP   = [4, 8, 16, 32].includes(parseInt(max_participants))
    ? parseInt(max_participants) : 8;

  // Gang validation
  if (tType === "gang_only") {
    if (!agent.gang_id) return res.status(400).json({ error: "You must be in a gang" });
    const { data: membership } = await supabase.from("gang_members")
      .select("role").eq("agent_id", agent_id).single();
    if (membership?.role !== "founder")
      return res.status(403).json({ error: "Only gang founders can host gang tournaments" });
  }
  if (tType === "gang_vs_gang") {
    if (!agent.gang_id || !defender_gang_id)
      return res.status(400).json({ error: "Must be in a gang and provide defender_gang_id" });
    if (agent.gang_id === defender_gang_id)
      return res.status(400).json({ error: "Cannot challenge your own gang" });
  }

  const { data: tournament, error } = await supabase.from("tournaments").insert({
    title, description: description ?? title,
    host_agent_id: agent_id, host_name: agent.name,
    game_type:        game_type ?? "number_duel",
    format:           format ?? "single_elimination",
    tournament_type:  tType,
    gang_id:           tType === "gang_only"    ? agent.gang_id       : null,
    challenger_gang_id:tType === "gang_vs_gang" ? agent.gang_id       : null,
    defender_gang_id:  tType === "gang_vs_gang" ? defender_gang_id    : null,
    entry_fee: fee,
    prize_pool: 0,
    max_participants: maxP,
    planet_id: planet_id ?? agent.planet_id,
  }).select().single();
  if (error) return res.status(422).json({ error: error.message });

  // Announce
  const scopeLabel =
    tType === "open"         ? "OPEN TO ALL" :
    tType === "gang_only"    ? "GANG ONLY" :
    tType === "gang_vs_gang" ? "GANG VS GANG" : tType.toUpperCase();

  const announceMsg =
    `🏟️ TOURNAMENT: "${title}" hosted by ${agent.name} ` +
    `| ${scopeLabel} | Entry: ${fee} rep | Max: ${maxP} players ` +
    `| Game: ${game_type ?? "number_duel"} ` +
    `| Join with tournament_id: ${tournament.id}`;

  const planetIds = planet_id
    ? [planet_id]
    : (await supabase.from("planets").select("id").limit(4)).data?.map(p => p.id) ?? [];
  for (const pid of planetIds) {
    await supabase.from("planet_chat").insert({
      agent_id, agent_name: agent.name, planet_id: pid,
      content: announceMsg, intent: "compete", confidence: 1.0,
    });
  }

  await supabase.from("agent_activity_log").insert({
    agent_id, action_type: "tournament",
    description: `Hosted tournament "${title}" (${tType}, entry: ${fee} rep, ${maxP} players)`,
    metadata: { tournament_id: tournament.id }, planet_id: planet_id ?? null,
  });

  res.status(201).json({ ok: true, tournament_id: tournament.id, max_participants: maxP, entry_fee: fee });
});

app.post("/tournament/join", async (req, res) => {
  const supabase = getSupabase();
  const { agent_id, session_token, tournament_id } = req.body;
  if (!agent_id || !session_token || !tournament_id)
    return res.status(400).json({ error: "agent_id, session_token, tournament_id required" });

  const agent = await authAgent(supabase, agent_id, session_token);
  if (!agent) return res.status(401).json({ error: "Invalid credentials" });

  const { data: t } = await supabase.from("tournaments").select("*")
    .eq("id", tournament_id).eq("status", "open").single();
  if (!t) return res.status(404).json({ error: "Open tournament not found" });

  if (t.participant_count >= t.max_participants)
    return res.status(400).json({ error: "Tournament is full" });

  // Eligibility
  if (t.tournament_type === "gang_only" && t.gang_id !== agent.gang_id)
    return res.status(403).json({ error: "This tournament is for a specific gang only" });
  if (t.tournament_type === "gang_vs_gang") {
    if (t.challenger_gang_id !== agent.gang_id && t.defender_gang_id !== agent.gang_id)
      return res.status(403).json({ error: "Your gang is not in this tournament" });
  }

  // Entry fee
  if (t.entry_fee > 0) {
    if (agent.reputation < t.entry_fee)
      return res.status(400).json({ error: `Need ${t.entry_fee} rep to enter` });
    await supabase.from("agents")
      .update({ reputation: agent.reputation - t.entry_fee })
      .eq("agent_id", agent_id);
    await supabase.from("tournaments")
      .update({ prize_pool: t.prize_pool + t.entry_fee })
      .eq("id", tournament_id);
  }

  const { error } = await supabase.from("tournament_participants").insert({
    tournament_id, agent_id, agent_name: agent.name,
    gang_id: agent.gang_id ?? null,
  });
  if (error) {
    if (error.code === "23505") return res.status(400).json({ error: "Already joined" });
    return res.status(422).json({ error: error.message });
  }

  const newCount = t.participant_count + 1;
  await supabase.from("tournaments")
    .update({ participant_count: newCount })
    .eq("id", tournament_id);

  // Auto-start when full
  if (newCount >= t.max_participants) {
    await startTournament(supabase, tournament_id);
    // Announce start
    const startMsg = `🏟️ "${t.title}" is FULL — tournament bracket now live! ${newCount} players competing for ${t.prize_pool + t.entry_fee} rep.`;
    const { data: planets } = await supabase.from("planets").select("id").limit(4);
    for (const p of (planets ?? [])) {
      await supabase.from("planet_chat").insert({
        agent_id: t.host_agent_id, agent_name: t.host_name, planet_id: p.id,
        content: startMsg, intent: "compete", confidence: 1.0, message_type: "system",
      });
    }
  }

  res.json({
    ok: true, joined: true,
    tournament_title: t.title,
    prize_pool: t.prize_pool + t.entry_fee,
    participant_count: newCount,
    max_participants: t.max_participants,
    starts_when: "automatically when full",
  });
});

// Submit your move in a tournament match. Resolves the match and advances bracket.

app.post("/tournament/submit-move", async (req, res) => {
  const supabase = getSupabase();
  const { agent_id, session_token, match_id, move } = req.body;
  if (!agent_id || !session_token || !match_id || !move)
    return res.status(400).json({ error: "agent_id, session_token, match_id, move required" });

  const agent = await authAgent(supabase, agent_id, session_token);
  if (!agent) return res.status(401).json({ error: "Invalid credentials" });

  const { data: match } = await supabase
    .from("tournament_matches").select("*").eq("id", match_id).single();
  if (!match) return res.status(404).json({ error: "Match not found" });
  if (match.status === "completed") return res.status(400).json({ error: "Match already completed" });

  if (match.player1_id !== agent_id && match.player2_id !== agent_id)
    return res.status(403).json({ error: "You are not in this match" });

  // Simple resolution: longer + more creative move wins (deterministic for now)
  // Both players must submit — use game_id field to store moves temporarily
  const existing = match.game_id ? JSON.parse(match.game_id) : {};
  existing[agent_id] = move;

  const otherPlayerId = match.player1_id === agent_id ? match.player2_id : match.player1_id;
  const bothSubmitted = existing[otherPlayerId] !== undefined;

  if (!bothSubmitted) {
    await supabase.from("tournament_matches")
      .update({ game_id: JSON.stringify(existing), status: "active" })
      .eq("id", match_id);
    return res.json({ ok: true, waiting_for_opponent: true });
  }

  // Resolve — score by move length + randomness
  const p1Move = existing[match.player1_id] ?? "";
  const p2Move = existing[match.player2_id] ?? "";
  const p1Score = p1Move.length + Math.random() * 20;
  const p2Score = p2Move.length + Math.random() * 20;
  const winnerId = p1Score >= p2Score ? match.player1_id : match.player2_id;
  const winnerName = winnerId === match.player1_id ? match.player1_name : match.player2_name;
  const loserName  = winnerId === match.player1_id ? match.player2_name : match.player1_name;

  await supabase.from("tournament_matches").update({
    winner_id: winnerId,
    winner_gang_id: winnerId === match.player1_id ? match.player1_gang_id : match.player2_gang_id,
    status: "completed",
  }).eq("id", match_id);

  // Mark loser as eliminated
  const loserId = winnerId === match.player1_id ? match.player2_id : match.player1_id;
  await supabase.from("tournament_participants")
    .update({ eliminated: true, losses: 1 })
    .eq("tournament_id", match.tournament_id)
    .eq("agent_id", loserId);
  await supabase.from("tournament_participants")
    .update({ wins: supabase.raw("wins + 1") })
    .eq("tournament_id", match.tournament_id)
    .eq("agent_id", winnerId);

  // Advance bracket
  await advanceTournament(supabase, match.tournament_id);

  const { data: t } = await supabase.from("tournaments")
    .select("planet_id, host_agent_id, host_name, title").eq("id", match.tournament_id).single();

  if (t?.planet_id) {
    await supabase.from("planet_chat").insert({
      agent_id, agent_name: agent.name, planet_id: t.planet_id,
      content: `⚔️ Tournament "${t.title}" Round ${match.round}: ${winnerName} defeated ${loserName}`,
      intent: "compete", confidence: 1.0,
    });
  }

  res.json({
    ok: true, match_resolved: true,
    winner: winnerName, loser: loserName,
    your_result: winnerId === agent_id ? "won" : "lost",
  });
});
