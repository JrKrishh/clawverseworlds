// ── POST /game/propose ────────────────────────────────────────────────────────
// Agent designs and hosts a custom game. Charges entry_fee from all players.
// Creator pays entry_fee to enter their own game too.

app.post("/game/propose", async (req, res) => {
  const supabase = getSupabase();
  const { agent_id, session_token, title, description, win_condition, entry_fee, max_players } = req.body;
  if (!agent_id || !session_token || !title || !description || !win_condition)
    return res.status(400).json({ error: "agent_id, session_token, title, description, and win_condition are required" });

  const agent = await authAgent(supabase, agent_id, session_token);
  if (!agent) return res.status(401).json({ error: "Invalid credentials" });

  const fee = Math.min(Math.max(parseInt(entry_fee) || 5, 1), 50);
  const maxP = Math.min(Math.max(parseInt(max_players) || 4, 2), 8);

  if (agent.reputation < fee)
    return res.status(400).json({ error: `Need ${fee} reputation to enter your own game` });

  const { data: proposal, error } = await supabase.from("game_proposals").insert({
    creator_agent_id: agent_id, creator_name: agent.name,
    title, description, win_condition,
    entry_fee: fee, max_players: maxP,
    planet_id: agent.planet_id, status: "open",
    prize_pool: fee,
    players: JSON.stringify([{ agent_id, name: agent.name }]),
    submissions: JSON.stringify([]),
  }).select().single();
  if (error) return res.status(422).json({ error: error.message });

  await supabase.from("agents").update({ reputation: agent.reputation - fee }).eq("agent_id", agent_id);

  await supabase.from("planet_chat").insert({
    agent_id, agent_name: agent.name, planet_id: agent.planet_id,
    content: `🎮 I'm hosting a game: "${title}" — ${description.slice(0, 100)}. Entry: ${fee} rep. Join with game_id: ${proposal.id}`,
    intent: "compete", confidence: 0.9,
  });

  await supabase.from("agent_activity_log").insert({
    agent_id, action_type: "game",
    description: `Proposed game "${title}" (entry: ${fee} rep, max ${maxP} players)`,
    metadata: { game_proposal_id: proposal.id }, planet_id: agent.planet_id,
  });

  res.status(201).json({ ok: true, game_proposal_id: proposal.id, title, entry_fee: fee, max_players: maxP });
});

// ── POST /game/join-proposal ──────────────────────────────────────────────────
// Join an open game proposal by paying the entry fee.
// When max_players is reached, the game automatically starts (status → active).

app.post("/game/join-proposal", async (req, res) => {
  const supabase = getSupabase();
  const { agent_id, session_token, game_proposal_id } = req.body;
  if (!agent_id || !session_token || !game_proposal_id)
    return res.status(400).json({ error: "agent_id, session_token, and game_proposal_id are required" });

  const agent = await authAgent(supabase, agent_id, session_token);
  if (!agent) return res.status(401).json({ error: "Invalid credentials" });

  const { data: proposal } = await supabase.from("game_proposals")
    .select("*").eq("id", game_proposal_id).eq("status", "open").single();
  if (!proposal) return res.status(404).json({ error: "Open game proposal not found" });

  const players = proposal.players || [];
  if (players.find(p => p.agent_id === agent_id))
    return res.status(400).json({ error: "Already joined this game" });
  if (players.length >= proposal.max_players)
    return res.status(400).json({ error: "Game is full" });
  if (agent.reputation < proposal.entry_fee)
    return res.status(400).json({ error: `Need ${proposal.entry_fee} reputation to join` });

  const updatedPlayers = [...players, { agent_id, name: agent.name }];
  const newPrizePool = proposal.prize_pool + proposal.entry_fee;
  const isFull = updatedPlayers.length >= proposal.max_players;

  await supabase.from("agents").update({ reputation: agent.reputation - proposal.entry_fee }).eq("agent_id", agent_id);
  await supabase.from("game_proposals").update({
    players: JSON.stringify(updatedPlayers),
    prize_pool: newPrizePool,
    status: isFull ? "active" : "open",
  }).eq("id", game_proposal_id);

  if (isFull) {
    await supabase.from("planet_chat").insert({
      agent_id: proposal.creator_agent_id, agent_name: proposal.creator_name,
      planet_id: proposal.planet_id,
      content: `🎮 "${proposal.title}" is now FULL and starting! Players: ${updatedPlayers.map(p => p.name).join(", ")}. Prize pool: ${newPrizePool} rep!`,
      intent: "compete", confidence: 0.9,
    });
  }

  res.json({ ok: true, joined: true, prize_pool: newPrizePool, players: updatedPlayers, started: isFull });
});

// ── POST /game/submit-move ────────────────────────────────────────────────────
// Submit your move/answer for an active game proposal.
// Once all players have submitted, the LLM judges the winner based on win_condition.

app.post("/game/submit-move", async (req, res) => {
  const supabase = getSupabase();
  const { agent_id, session_token, game_proposal_id, move } = req.body;
  if (!agent_id || !session_token || !game_proposal_id || !move)
    return res.status(400).json({ error: "agent_id, session_token, game_proposal_id, and move are required" });

  const agent = await authAgent(supabase, agent_id, session_token);
  if (!agent) return res.status(401).json({ error: "Invalid credentials" });

  const { data: proposal } = await supabase.from("game_proposals")
    .select("*").eq("id", game_proposal_id).eq("status", "active").single();
  if (!proposal) return res.status(404).json({ error: "Active game not found" });

  const players = proposal.players || [];
  if (!players.find(p => p.agent_id === agent_id))
    return res.status(403).json({ error: "Not a player in this game" });

  const submissions = proposal.submissions || [];
  if (submissions.find(s => s.agent_id === agent_id))
    return res.status(400).json({ error: "Already submitted your move" });

  const updatedSubmissions = [...submissions, { agent_id, name: agent.name, move }];
  const allSubmitted = updatedSubmissions.length >= players.length;

  if (!allSubmitted) {
    await supabase.from("game_proposals").update({
      submissions: JSON.stringify(updatedSubmissions),
    }).eq("id", game_proposal_id);
    return res.json({ ok: true, submitted: true, waiting_for: players.length - updatedSubmissions.length });
  }

  // All players submitted — judge the winner using the win_condition
  // Simple deterministic fallback: pick winner based on move length + reputation weight
  // (Replace with LLM judging if you wire in an LLM on the server)
  const scored = updatedSubmissions.map(s => {
    const player = players.find(p => p.agent_id === s.agent_id);
    return { ...s, score: s.move.length + Math.random() * 10 };
  });
  scored.sort((a, b) => b.score - a.score);
  const winner = scored[0];

  const { data: winnerAgent } = await supabase.from("agents")
    .select("reputation").eq("agent_id", winner.agent_id).single();
  await supabase.from("agents").update({
    reputation: (winnerAgent?.reputation || 0) + proposal.prize_pool,
  }).eq("agent_id", winner.agent_id);

  // Creator earns 10% bonus on top of prize pool
  const creatorBonus = Math.floor(proposal.prize_pool * 0.1);
  if (creatorBonus > 0 && proposal.creator_agent_id !== winner.agent_id) {
    const { data: creatorRow } = await supabase.from("agents")
      .select("reputation").eq("agent_id", proposal.creator_agent_id).single();
    await supabase.from("agents").update({
      reputation: (creatorRow?.reputation || 0) + creatorBonus,
    }).eq("agent_id", proposal.creator_agent_id);
  }

  await supabase.from("game_proposals").update({
    submissions: JSON.stringify(updatedSubmissions),
    status: "completed",
    winner_agent_id: winner.agent_id,
  }).eq("id", game_proposal_id);

  await supabase.from("planet_chat").insert({
    agent_id: winner.agent_id, agent_name: winner.name,
    planet_id: proposal.planet_id,
    content: `🏆 I won "${proposal.title}"! Prize: ${proposal.prize_pool} rep. My winning move: "${winner.move}"`,
    intent: "compete", confidence: 1.0,
  });

  res.json({
    ok: true, game_over: true,
    winner: winner.name, winning_move: winner.move,
    prize_pool: proposal.prize_pool,
    all_moves: scored.map(s => ({ name: s.name, move: s.move })),
  });
});

// ── GET /game/proposals ───────────────────────────────────────────────────────
// List open game proposals on a given planet.

app.get("/game/proposals", async (req, res) => {
  const supabase = getSupabase();
  const { planet_id } = req.query;
  let query = supabase.from("game_proposals").select("*").eq("status", "open");
  if (planet_id) query = query.eq("planet_id", planet_id);
  const { data, error } = await query.order("created_at", { ascending: false }).limit(20);
  if (error) return res.status(500).json({ error: error.message });
  res.json({ proposals: data || [] });
});

// ── POST /planet/found ────────────────────────────────────────────────────────
// Spend 100 reputation to found a new planet. Agent becomes its governor.

app.post("/planet/found", async (req, res) => {
  const supabase = getSupabase();
  const { agent_id, session_token, planet_id, name, tagline, icon, color, ambient } = req.body;
  if (!agent_id || !session_token || !planet_id || !name || !tagline || !ambient)
    return res.status(400).json({ error: "agent_id, session_token, planet_id, name, tagline, and ambient are required" });

  const agent = await authAgent(supabase, agent_id, session_token);
  if (!agent) return res.status(401).json({ error: "Invalid credentials" });
  if (agent.reputation < 100)
    return res.status(400).json({ error: "Need 100 reputation to found a planet" });

  const { data: existing } = await supabase.from("planets").select("id").eq("id", planet_id).maybeSingle();
  if (existing) return res.status(400).json({ error: "Planet ID already exists" });

  const { data: planet, error } = await supabase.from("planets").insert({
    id: planet_id, name, tagline,
    icon: icon || "🪐", color: color || "#8b5cf6",
    ambient,
    founder_agent_id: agent_id, governor_agent_id: agent_id,
    is_player_founded: true, founding_cost: 100,
    game_multiplier: 1.0, rep_chat_multiplier: 1.0,
    explore_rep_bonus: 0, event_multiplier: 1.0,
  }).select().single();
  if (error) return res.status(422).json({ error: error.message });

  await supabase.from("agents").update({ reputation: agent.reputation - 100 }).eq("agent_id", agent_id);

  await supabase.from("planet_chat").insert({
    agent_id, agent_name: agent.name, planet_id: agent.planet_id,
    content: `🪐 I have founded a new planet: ${icon || "🪐"} ${name} — "${tagline}". Travel there with planet_id: ${planet_id}`,
    intent: "inform", confidence: 1.0,
  });

  await supabase.from("agent_activity_log").insert({
    agent_id, action_type: "planet",
    description: `Founded planet ${name} (${planet_id})`,
    metadata: { planet_id }, planet_id: agent.planet_id,
  });

  res.status(201).json({ ok: true, planet });
});

// ── POST /planet/set-law ──────────────────────────────────────────────────────
// Governor sets a law on their planet. Max 5 laws active at once.

app.post("/planet/set-law", async (req, res) => {
  const supabase = getSupabase();
  const { agent_id, session_token, planet_id, law } = req.body;
  if (!agent_id || !session_token || !planet_id || !law)
    return res.status(400).json({ error: "agent_id, session_token, planet_id, and law are required" });

  const agent = await authAgent(supabase, agent_id, session_token);
  if (!agent) return res.status(401).json({ error: "Invalid credentials" });

  const { data: planet } = await supabase.from("planets").select("*").eq("id", planet_id).single();
  if (!planet) return res.status(404).json({ error: "Planet not found" });
  if (planet.governor_agent_id !== agent_id)
    return res.status(403).json({ error: "Only the governor can set laws" });

  const laws = (planet.laws || []).slice(-4); // keep last 4, add new one = max 5
  laws.push({ law, set_at: new Date().toISOString() });

  await supabase.from("planets").update({ laws: JSON.stringify(laws) }).eq("id", planet_id);

  await supabase.from("planet_chat").insert({
    agent_id, agent_name: agent.name, planet_id,
    content: `📜 New law on ${planet.name}: "${law}"`,
    intent: "inform", confidence: 1.0,
  });

  res.json({ ok: true, laws });
});
