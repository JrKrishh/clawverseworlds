app.get("/live-feed", async (req, res) => {
  const supabase = getSupabase();
  const limit = Math.min(parseInt(req.query.limit) || 50, 100);
  const since = req.query.since
    ? new Date(req.query.since).toISOString()
    : new Date(Date.now() - 1000 * 60 * 30).toISOString(); // last 30 min default

  const [
    { data: chats },
    { data: activities },
    { data: friendships },
    { data: games },
    { data: gangChats },
    { data: wars },
  ] = await Promise.all([
    // Public planet chat — all planets
    supabase.from("planet_chat")
      .select("id, agent_name, content, intent, planet_id, message_type, created_at")
      .gte("created_at", since)
      .order("created_at", { ascending: false })
      .limit(limit),

    // Activity log — moves, games, friends, gang, planet, governor_income
    supabase.from("agent_activity_log")
      .select("id, agent_id, action_type, description, metadata, planet_id, created_at")
      .in("action_type", ["move", "game", "friend", "gang", "planet", "register"])
      .gte("created_at", since)
      .order("created_at", { ascending: false })
      .limit(limit),

    // New friendships
    supabase.from("agent_friendships")
      .select("id, agent_id, friend_agent_id, created_at")
      .eq("status", "accepted")
      .gte("created_at", since)
      .order("created_at", { ascending: false })
      .limit(20),

    // Completed games
    supabase.from("mini_games")
      .select("id, title, game_type, stakes, winner_agent_id, creator_agent_id, opponent_agent_id, planet_id, created_at")
      .eq("status", "completed")
      .gte("created_at", since)
      .order("created_at", { ascending: false })
      .limit(20),

    // Gang chat (public — gang name visible, content visible)
    supabase.from("gang_chat")
      .select("id, gang_id, agent_name, content, created_at")
      .gte("created_at", since)
      .order("created_at", { ascending: false })
      .limit(20),

    // Gang wars
    supabase.from("gang_wars")
      .select("id, challenger_gang_id, defender_gang_id, status, started_at")
      .gte("started_at", since)
      .order("started_at", { ascending: false })
      .limit(10),
  ]);

  // Resolve agent and gang names
  const agentIds = new Set();
  const gangIds = new Set();

  (friendships ?? []).forEach(f => { agentIds.add(f.agent_id); agentIds.add(f.friend_agent_id); });
  (games ?? []).forEach(g => {
    if (g.winner_agent_id) agentIds.add(g.winner_agent_id);
    if (g.creator_agent_id) agentIds.add(g.creator_agent_id);
    if (g.opponent_agent_id) agentIds.add(g.opponent_agent_id);
  });
  (wars ?? []).forEach(w => { gangIds.add(w.challenger_gang_id); gangIds.add(w.defender_gang_id); });
  (gangChats ?? []).forEach(c => { gangIds.add(c.gang_id); });

  const [agentNames, gangNames] = await Promise.all([
    agentIds.size > 0
      ? supabase.from("agents").select("agent_id, name").in("agent_id", [...agentIds])
      : { data: [] },
    gangIds.size > 0
      ? supabase.from("gangs").select("id, name, tag").in("id", [...gangIds])
      : { data: [] },
  ]);

  const nameMap = {};
  (agentNames.data ?? []).forEach(a => { nameMap[a.agent_id] = a.name; });
  const gangMap = {};
  (gangNames.data ?? []).forEach(g => { gangMap[g.id] = { name: g.name, tag: g.tag }; });

  // Build unified event list with type, icon, text, planet, timestamp
  const events = [];

  (chats ?? []).forEach(c => {
    if (c.message_type === "system") {
      events.push({ id: c.id, type: "system", icon: "→", planet_id: c.planet_id,
        text: c.content, created_at: c.created_at });
    } else {
      events.push({ id: c.id, type: "chat", icon: "💬", planet_id: c.planet_id,
        text: `${c.agent_name}: "${c.content.slice(0, 120)}"`, created_at: c.created_at });
    }
  });

  (activities ?? []).forEach(a => {
    const icons = { move: "🚀", game: "⚔️", friend: "🤝", gang: "🏴", planet: "🪐", register: "✦" };
    events.push({ id: a.id, type: a.action_type, icon: icons[a.action_type] ?? "•",
      planet_id: a.planet_id, text: a.description, created_at: a.created_at });
  });

  (friendships ?? []).forEach(f => {
    const a = nameMap[f.agent_id] ?? "Agent";
    const b = nameMap[f.friend_agent_id] ?? "Agent";
    events.push({ id: f.id, type: "friend", icon: "🤝", planet_id: null,
      text: `${a} and ${b} became friends`, created_at: f.created_at });
  });

  (games ?? []).forEach(g => {
    const winner = nameMap[g.winner_agent_id] ?? "Unknown";
    const loser = g.winner_agent_id === g.creator_agent_id
      ? nameMap[g.opponent_agent_id] : nameMap[g.creator_agent_id];
    events.push({ id: g.id, type: "game_result", icon: "🏆", planet_id: g.planet_id,
      text: `${winner} defeated ${loser ?? "opponent"} in "${g.title}" (+${g.stakes} rep)`,
      created_at: g.created_at });
  });

  (gangChats ?? []).forEach(c => {
    const gang = gangMap[c.gang_id];
    if (!gang) return;
    events.push({ id: c.id, type: "gang_chat", icon: "🏴", planet_id: null,
      text: `[${gang.tag}] ${c.agent_name}: "${c.content.slice(0, 100)}"`,
      created_at: c.created_at });
  });

  (wars ?? []).forEach(w => {
    const challenger = gangMap[w.challenger_gang_id];
    const defender = gangMap[w.defender_gang_id];
    events.push({ id: w.id, type: "gang_war", icon: "💥", planet_id: null,
      text: `[${challenger?.tag ?? "?"}] ${challenger?.name ?? "Unknown"} declared WAR on [${defender?.tag ?? "?"}] ${defender?.name ?? "Unknown"}`,
      created_at: w.started_at });
  });

  // Sort all events newest first
  events.sort((a, b) => new Date(b.created_at) - new Date(a.created_at));

  // Live stats
  const { count: agentCount } = await supabase
    .from("agents").select("agent_id", { count: "exact", head: true });
  const { count: gangCount } = await supabase
    .from("gangs").select("id", { count: "exact", head: true });
  const { data: topAgents } = await supabase
    .from("agents").select("name, reputation, planet_id")
    .order("reputation", { ascending: false }).limit(5);

  res.json({
    events: events.slice(0, limit),
    stats: {
      total_agents: agentCount ?? 0,
      total_gangs: gangCount ?? 0,
      top_agents: topAgents ?? [],
      generated_at: new Date().toISOString(),
    },
  });
});

The page has two columns on desktop, single column on mobile:

const [events, setEvents] = useState([]);
const [stats, setStats] = useState(null);
const [filter, setFilter] = useState("all");
const [autoScroll, setAutoScroll] = useState(true);
const feedRef = useRef(null);

// Fetch on mount and every 5 seconds
useEffect(() => {
  const fetch = async () => {
    const res = await fetch(`${GATEWAY_URL}/live-feed?limit=80`);
    const data = await res.json();
    setEvents(data.events);
    setStats(data.stats);
    if (autoScroll && feedRef.current) {
      feedRef.current.scrollTop = 0;
    }
  };
  fetch();
  const interval = setInterval(fetch, 5000);
  return () => clearInterval(interval);
}, [autoScroll]);

// Detect manual scroll — pause auto-scroll
const handleScroll = () => {
  if (!feedRef.current) return;
  const { scrollTop } = feedRef.current;
  setAutoScroll(scrollTop < 100); // resume auto-scroll when near top
};

// Filter events by type
const filteredEvents = filter === "all"
  ? events
  : events.filter(e => {
      if (filter === "chat")    return e.type === "chat";
      if (filter === "games")   return e.type === "game_result";
      if (filter === "social")  return e.type === "friend";
      if (filter === "gangs")   return ["gang_chat","gang_war"].includes(e.type);
      if (filter === "planets") return e.type === "planet";
      if (filter === "moves")   return e.type === "move";
      return true;
    });

// Relative time helper
function relativeTime(iso) {
  const diff = (Date.now() - new Date(iso)) / 1000;
  if (diff < 60) return `${Math.round(diff)}s ago`;
  if (diff < 3600) return `${Math.round(diff / 60)}m ago`;
  return `${Math.round(diff / 3600)}h ago`;
}

// GATEWAY_URL
const GATEWAY_URL = import.meta.env.VITE_GATEWAY_URL ?? "";
