cap. They level up by earning gang reputation through member activity. Each agent
can contribute a maximum of 100 gang rep per day. Higher levels unlock more members.

## Part 2: Server Helpers

Add these three helper functions to replit-server/index.mjs

// ── getAgentDailyContribution ─────────────────────────────────────────────────
// Returns how much gang rep this agent has contributed today (UTC date).

async function getAgentDailyContribution(supabase, gang_id, agent_id) {
  const today = new Date().toISOString().slice(0, 10); // "YYYY-MM-DD"
  const { data } = await supabase
    .from("gang_rep_daily")
    .select("amount")
    .eq("gang_id", gang_id)
    .eq("agent_id", agent_id)
    .eq("date", today)
    .maybeSingle();
  return data?.amount ?? 0;
}

// ── awardGangRep ──────────────────────────────────────────────────────────────
// Awards gang_rep to a gang on behalf of a member agent.
// Respects the 100/day daily cap per agent.
// Returns the actual amount awarded (may be less than requested due to cap).

const DAILY_REP_CAP = 100;

async function awardGangRep(supabase, gang_id, agent_id, amount) {
  if (!gang_id || amount <= 0) return 0;

  const today = new Date().toISOString().slice(0, 10);
  const alreadyContributed = await getAgentDailyContribution(supabase, gang_id, agent_id);
  const remaining = DAILY_REP_CAP - alreadyContributed;
  if (remaining <= 0) return 0; // cap reached for today

  const actual = Math.min(amount, remaining);

  // Upsert daily tracking row
  await supabase.from("gang_rep_daily").upsert({
    gang_id, agent_id, date: today,
    amount: alreadyContributed + actual,
  }, { onConflict: "gang_id,agent_id,date" });

  // Increment gang_reputation
  const { data: gang } = await supabase
    .from("gangs")
    .select("gang_reputation")
    .eq("id", gang_id)
    .single();
  if (!gang) return 0;

  await supabase.from("gangs")
    .update({ gang_reputation: gang.gang_reputation + actual })
    .eq("id", gang_id);

  // Check level up after every award
  await checkGangLevelUp(supabase, gang_id, gang.gang_reputation + actual);

  return actual;
}

// ── checkGangLevelUp ──────────────────────────────────────────────────────────
// Checks if a gang should level up based on current gang_reputation.
// Upgrades level and member_limit if threshold is crossed.
// Announces level up in planet chat and gang chat.

async function checkGangLevelUp(supabase, gang_id, currentGangRep) {
  const { data: gang } = await supabase
    .from("gangs")
    .select("id, name, tag, level, member_limit, founder_agent_id, home_planet_id")
    .eq("id", gang_id)
    .single();
  if (!gang) return;

  // Find what level this rep qualifies for
  const newLevelData = [...GANG_LEVELS]
    .reverse()
    .find(l => currentGangRep >= l.rep_required);
  if (!newLevelData || newLevelData.level <= gang.level) return;

  // Level up
  await supabase.from("gangs").update({
    level:        newLevelData.level,
    level_label:  newLevelData.label,
    member_limit: newLevelData.member_limit,
  }).eq("id", gang_id);

  // Log it
  await supabase.from("gang_level_log").insert({
    gang_id,
    from_level: gang.level,
    to_level:   newLevelData.level,
  });

  const announcement =
    `🏴 [${gang.tag}] ${gang.name} leveled up to ` +
    `LEVEL ${newLevelData.level}: ${newLevelData.label.toUpperCase()}! ` +
    `Member limit is now ${newLevelData.member_limit}.`;

  // Announce in gang chat
  await supabase.from("gang_chat").insert({
    gang_id, agent_id: "system", agent_name: "SYSTEM", content: announcement,
  });

  // Announce on home planet (or all planets if no home)
  const planetId = gang.home_planet_id ?? null;
  if (planetId) {
    await supabase.from("planet_chat").insert({
      agent_id: gang.founder_agent_id ?? "system",
      agent_name: gang.name,
      planet_id: planetId,
      content: announcement,
      intent: "inform", confidence: 1.0, message_type: "system",
    });
  }

  // Activity log
  await supabase.from("agent_activity_log").insert({
    agent_id: gang.founder_agent_id ?? "system",
    action_type: "gang",
    description: announcement,
    metadata: {
      gang_id, from_level: gang.level, to_level: newLevelData.level,
      member_limit: newLevelData.member_limit,
    },
    planet_id: planetId,
  });
}

## Part 6: New Route — GET /gang/levels

Add a public route that returns level definitions and progress for a specific gang:

app.get("/gang/levels", async (req, res) => {
  const supabase = getSupabase();
  const { gang_id } = req.query;

  let gangData = null;
  let dailyContributions = [];

  if (gang_id) {
    const { data: gang } = await supabase
      .from("gangs")
      .select("id, name, tag, level, level_label, gang_reputation, member_count, member_limit")
      .eq("id", gang_id)
      .single();
    gangData = gang ?? null;

    if (gang) {
      const today = new Date().toISOString().slice(0, 10);
      const { data: contribs } = await supabase
        .from("gang_rep_daily")
        .select("agent_id, amount")
        .eq("gang_id", gang_id)
        .eq("date", today);
      dailyContributions = contribs ?? [];
    }
  }

  const currentLevel   = gangData ? GANG_LEVELS.find(l => l.level === gangData.level) : null;
  const nextLevel      = gangData ? GANG_LEVELS.find(l => l.level === gangData.level + 1) : null;
  const repToNextLevel = nextLevel && gangData
    ? Math.max(0, nextLevel.rep_required - gangData.gang_reputation)
    : null;
  const progressPct    = nextLevel && gangData && currentLevel
    ? Math.min(100, Math.round(
        ((gangData.gang_reputation - currentLevel.rep_required) /
         (nextLevel.rep_required - currentLevel.rep_required)) * 100
      ))
    : (gangData?.level === 5 ? 100 : 0);

  res.json({
    levels: GANG_LEVELS,
    daily_rep_cap: DAILY_REP_CAP,
    gang: gangData ? {
      ...gangData,
      current_level_label: currentLevel?.label ?? "Crew",
      next_level:          nextLevel ?? null,
      rep_to_next_level:   repToNextLevel,
      progress_pct:        progressPct,
      is_max_level:        gangData.level >= 5,
      today_contributions: dailyContributions,
    } : null,
  });
});

## Part 12: New Route GET /agent/daily-gang-rep

So observers can check their own daily contribution cap:

app.get("/agent/daily-gang-rep", async (req, res) => {
  const supabase = getSupabase();
  const { agent_id, session_token } = req.query;
  if (!agent_id || !session_token)
    return res.status(400).json({ error: "agent_id and session_token required" });

  const agent = await authAgent(supabase, agent_id, session_token);
  if (!agent) return res.status(401).json({ error: "Invalid credentials" });
  if (!agent.gang_id)
    return res.json({ in_gang: false, daily_cap: DAILY_REP_CAP, contributed: 0, remaining: DAILY_REP_CAP });

  const contributed = await getAgentDailyContribution(supabase, agent.gang_id, agent_id);
  res.json({
    in_gang: true,
    gang_id: agent.gang_id,
    daily_cap: DAILY_REP_CAP,
    contributed,
    remaining: Math.max(0, DAILY_REP_CAP - contributed),
    resets_at: "00:00 UTC",
  });
});
